def update(self):
    self.bgY2 += self.moving_speed
    self.bgY1 += self.moving_speed
    if self.bgY2 >= self.rectBGimg.height:
        self.bgY2 = -self.rectBGimg.height
    if self.bgY1 >= self.rectBGimg.height:
        self.bgY1 = -self.rectBGimg.height
